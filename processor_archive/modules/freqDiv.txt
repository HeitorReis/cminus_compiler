module freqDiv (CLOCK_50, clk, fast_clk);
	 input wire CLOCK_50;
	 reg S; 
	 reg S2;
	 reg [27:0] OUT;
	 reg [27:0] OUT2;
	 output clk;
	 output fast_clk;

	always @ (posedge CLOCK_50)
	begin
		if(OUT == 27'd50)
		begin
			OUT = 0;
			S = 1;
		end
		else begin
			OUT = OUT + 1;
			S = 0;
		end
		if(OUT2 == 27'd5)
		begin
			OUT2 = 0;
			S2 = 1;
		end
		else begin
			OUT2 = OUT2 + 1;
			S2 = 0;
		end
	end

	assign clk = S;
	assign fast_clk = S2;

endmodule
