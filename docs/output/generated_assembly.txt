.text
.global main

main:
	bl: input
	mov: r4 = r0
	storei: [var_x] = r4
	loadi: r5 = [var_x]
	muli: r6 = r5, 2
	storei: [var_y] = r6
	loadi: r7 = [var_y]
	mov: r0 = r7
	bl: output
	movi: r8 = 1
	mov: r0 = r8
	bi: main_epilogue
main_epilogue:
	ret:

.data
var_input: .word 0
var_output: .word 0
var_x: .word 0
var_y: .word 0
